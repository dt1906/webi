/* Write a C++ program implementing a class with the name Circle having two functions with the 
   names:GetRadius and CalArea.The functions should get the value for the radius from the user
   and then calculate the area of the circle and display the results. */
#include<iostream>
using namespace std;
class Circle{
	private:
		int radius,area;
	public:
		void GetRadius()
		{
			cout<<"Enter the radius:";
			cin>>radius;
		}
		void CalArea()
		{
			area=3.14*radius*radius;
			cout<<"AREA OF CIRCLE IS:"<<area;
		}
};
int main()
{
	Circle o1;
	o1.GetRadius();
	o1.CalArea();
	return 0;
}
